<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Google Spreadsheet Data Weergeven</title>

    <!-- Laad de Google Charts API -->

    <script type="text/javascript" src="https://www.gstatic.com/charts/loader.js"></script>

    <script type="text/javascript">

        // Laad de benodigde visualisatiepakket

        google.charts.load('current', {'packages':['table']});


        // Wanneer de API is geladen, roep de functie drawTable() aan

        google.charts.setOnLoadCallback(drawTable);


        function drawTable() {

            // Maak een query naar de Google Spreadsheet

            var query = new google.visualization.Query(

                'https://docs.google.com/spreadsheets/d/1MBXz_WnQ38-0MWXIef1pUZ1lTUvynUnNZOdosXkTcDY/gviz/tq?gid=1525579811&range=B3:C5');


            // Verstuur de query en roep de response handler aan

            query.send(handleQueryResponse);

        }


        function handleQueryResponse(response) {

            // Controleer op fouten in de respons

            if (response.isError()) {

                alert('Fout bij het ophalen van gegevens: ' + response.getMessage());

                return;

            }


            // Verkrijg de gegevens als een DataTable

            var data = response.getDataTable();


            // Maak een nieuwe tabel en teken de gegevens in het element met id="table_div"

            var table = new google.visualization.Table(document.getElementById('table_div'));


            // Teken de tabel met gewenste opties

            table.draw(data, {

                showRowNumber: true,   // Toon rij-nummers

                width: '400px',        // Breedte van de tabel

                height: 'auto'         // Automatische hoogte

            });

        }

    </script>

</head>

<body>

    <h1>Gegevens van Google Spreadsheet</h1>

    <!-- Div waarin de tabel wordt getekend -->

    <div id="table_div"></div>

</body>
